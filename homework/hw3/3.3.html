def loss(W, X, y, lam):
#     dW = np.zeros(W.shape, dtype=np.float32) # initialize the gradient as zero

    # compute the loss and the gradient
    num_classes = W.shape[1]
    num_train = X.shape[0]
    loss = 0.0
#     for i in range(num_train):
#         scores = X[i,:].dot(W)

#             if j == y[i]:
#                 continue
#             margin = scores[j] - correct_class_score + 1 
#             if margin > 0:
#                 loss += margin

    scores = X.dot(W)
    yi_scores = scores[np.arange(scores.shape[0]),y] # http://stackoverflow.com/a/23435843/459241 
    margins = np.maximum(0, scores - np.matrix(yi_scores).T + 1)
    margins[np.arange(num_train),y] = 0
    loss = np.mean(np.sum(margins, axis=1))
#     loss += 0.5 * reg * np.sum(W * W)


    # Averaging over all examples
#     loss /= num_train
    # dW /= num_train

#     r = 0;
    
#     for j in range(0, 784):
#         for c in range(0, 10):
#             sum += W[j][c] * W[j][c]
    r = np.sum(np.linalg.norm(W, axis = 1))

    # Add regularization
    loss += lam *  r
    # dW += reg*W
    return loss

def compute_g(W, X, y):
    i = np.random.randint(0, 60000)
    dW = np.zeros(W.shape)
    scores = X[i,:].dot(W)
    for j in range(0, num_classes):
        if j == y[i]:
            continue
        margin = scores[j] - scores[y[i]] + 1 
        if margin > 0:
            dW[:,y[i]] -= X[i,:] 
            dW[:,j] += X[i,:] 
    return dW

def compute_theta_new_old(W, dW, gamma, lam):
    H = W - gamma * dW
    theta_new = np.random.rand(784, 10)
    for j in range(0, 10):
        norm = np.linalg.norm(H[:,j], 2)
        if (norm == 0):
            theta_new[:, j] = 0
        theta_new[:, j] = max(0, (1 - (gamma * lam / np.linalg.norm(H[:, j], 2)))) * W[:, j]
    return theta_new

def compute_theta_new_b(W, dW, gamma, lam):
    H = W - gamma * dW
    norm = np.apply_along_axis(np.linalg.norm, 0, H)
    norm_I = 1./norm
    return np.multiply(H, (1 - gamma * lam * norm_I).clip(min = 0))

from sklearn.metrics import accuracy_score
lam = 10
W = np.random.randn(784, 10)
X_test = np.array(images_test)
y_test = np.array(labels_test)
output = []
for i in tqdm(range(1, 1000000)):
#     losss = loss(W, X, y, lam)
    
#     loss, g = svm_loss_vectorized(W, X, y, lam)
    g = compute_g(W, X, y)
#     print(np.nonzero(g))
    gamma = 1/i
    
    theta_new = compute_theta_new_b(W, g, gamma, lam)
    W = theta_new
    if (i % 1000 == 0):
        y_pred = np.argmax(X_test.dot(W), 1)
        output.append(accuracy_score(y_pred, y_test))

W_10 = W
plt.plot(output)
plt.xlabel('Iteration (x 1000)')
plt.ylabel('Accuracy')
plt.title('lambda = {}'.format(lam))
